class Solution {
    public int[] searchRange(int[] nums, int target) {
        int left = searchRangeMin(nums, target);
        if (left == -1) {
            return new int[]{-1, -1};
        }

        int right = searchRangeMax(nums, target);
        return new int[]{left, right};
    }

    private int searchRangeMin(int[] nums, int target) {
        int left = 0; 
        int right = nums.length;
        while (left < right) {
            int mid = left + (right - left) / 2;

            if (nums[mid] == target) {
                right = mid;
            } else if (nums[mid] < target) {
                left = mid + 1;
            } else {
                right = mid;
            }
        }

        if (left < 0 || left >= nums.length) {
            return -1;
        }

        return nums[left] == target ? left : -1;
    }

    private int searchRangeMax(int[] nums, int target) {
        int left = 0; 
        int right = nums.length;
        while (left < right) {
            int mid = left + (right - left) / 2;

            if (nums[mid] == target) {
                left = mid + 1;
            } else if (nums[mid] < target) {
                left = mid + 1;
            } else {
                right = mid;
            }
        }

        if (right - 1 < 0 || right > nums.length) {
            return -1;
        }

        return nums[right - 1] == target ? right - 1 : -1;
    }
}
